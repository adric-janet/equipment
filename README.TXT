Quickstart:

 mkdir lib
 mkdir var
 mkdir var/item
 mkdir htdocs/data
 mkdir htdocs/data/org
 cd lib
 git clone https://github.com/cgutteridge/Graphite.git
 git clone https://github.com/semsol/arc2.git
